༄༅། ཁྲེགས་གཅོད་བསྡུད་པ་བཞུགས་སྷོ།༄༄༄། །​ རྒྱ་གར་སྐད་དུ། ས་མ་བ་ཏཱན་མ། བོད་སྐད་དུ།  ཁྲེག་གཅོད་བདུད་པ་ཞེས་བྱ་བ། །​ བཅོམ་ལྡན་འདས་མཐའ་གཞི་རང་གྲོལ་བའི་མགོན་པོ་རྫོགས་སངས་རྒྱས་པའི་བླ་མ་ལ་ཕྱག་འཚལ་ལོ། །​ དེ་ནས་སྟོན་པ་རྡོ་རྗེ་འཆང་། །​ ཕྱོགས་བཅུ་འཇིག་རྟེན་རབ་འབྱམས་སུ། །​ ཞལ་གྱི་དཀྱིལ་འཁོར་གསལ་བྱས་པས། །​ ལྗགས་ཀྱི་དབང་པོའི་རྕེ་མོ་ལས། །​ འོད་ཟེར་བརྒྱ་ཕྲག་དུ་མ་འཕྲོས། །​ འོད་ཀྱི་ཟེར་ལས་སངས་རྒྱས་ནི། །​ ཉིད་དང་འདྲ་བའི་གཙོ་འཁོར་ནི། །​ སྟོང་ཕྲག་བརྗོད་ཀྱི་མི་ལང་བ། །​ དུ་མ་འཕྲོས་པའི་སེམས་ཅན་གྱིས། །​ དོན་བྱས་ཉིད་ཀྱི་ཐུགས་ཁར་འདུས། །​ རང་བཞིན་དགོངས་པ་ལྕོག་གཞག་ལས། །​ ཚིག་མིན་དོན་གྱི་དགོངས་པ་འཕྲོས། །​ ཀྱཱེཻ་མ་སེམས་དཔའ་ཆེན་པོ་ཉོན། །​ སྣང་བ་རང་གི་རང་བཅད་པས། །​ ཁྲེགས་གཅོད་རྣམས་ཀྱི་དང་པོར་བྱུང་། །​ བྱས་པ་མེད་པའི་ཆོས་ཉིད་ལ། །​ བྱས་བྱུང་པ་མེད་པའི་གནད་མཐོང་ན། །​ བྱས་པ་བྱུང་བ་རང་ས་ཆོད། །​ འཁོར་བ་ཞེན་མཐའ་རང་གནས་བས། །​ འཁྲུལ་པ་བདེན་པ་རང་གནས་ཏེ། །​ དུས་གསུམ་རྒྱལ་བའི་དགོངས་པ་རྙེད། །​ འདི་ལ་རབ་གནས་མཐའ་ཡལ་ཏེ། །​ ང་ཡིས་ཆུད་རྩའི་རོ་བརྒྱད་སྙེད། །​ ཐོག་མ་རང་དང་རང་བཞིན་ལས། །​ བདག་ཉིད་ཆེན་པོའི་ལྟས་སྟངས་ལས། །​ མཐོང་བ་མེད་པའི་ཡུལ་རྙེད་བ། །​བཤད་པར་བྱ་བའི་ཆོས་རྣམས་ཟད། །​ཐོས་བསམ་བསྒོམ་པའི་ཉེན་ལེན་སྟོངས། །​ དུ་མ་འདུས་པའི་ཚང་ཤིག་པས། །​ གཅིག་བར་གནས་པའི་ཡེ་ཤེས་མཐོང་། །​ དེ་མཐོང་བ་ཙམ་གྱིས་འཁོར་བའི་ལས། །​ མ་བྱས་གཞི་ལ་ཐམས་ཅད་འགྲུབ། །​ འདིས་ནི་བྱུང་བའི་བདག་པོ་ཟིན། །​ དེ་ལ་ཡོན་ཏན་བརྒྱ་ཕྲག་སྙེད། །​ དེ་སྙེད་ཆོས་ལ་འཇིག་པ་མེད། །​ འཇིགས་མེད་ལམ་དུ་སུ་ཞུགས་པ། །​ འདི་ལ་ལྡོག་པ་ཡོད་མི་སྲིད། །​ སྲིད་བྲལ་མིང་བཞི་སྟོངས་པའོ། །​ ཁྱབ་བདག་ནམ་མཁའི་མཐའ་སྙེད་པ། །​ དབྱིང་སུ་ཚུད་པས་ཕྱི་ནང་ཆོད། །​ ས་ཆུ་མེ་རླུང་འབྱུང་བ་ཡི། །​ གཞི་དང་ལམ་དང་འབྲས་བུ་ཟད། །​ ལྟ་དང་སྒོམ་དང་སྤྱོད་པ་གཞི། །​ རང་སར་དག་པས་འཁོར་རྒྱུན་ཆོད། །​ བདེ་སྡུག་སྙོམས་པས་དགའ་བདེ་ཡིས། །​ འདོད་བ་དག་པས་རང་འདོད་ཐོབས། །​ དེ་ལ་གང་གིས་སྙས་པ་མེད། །​ འདི་ནི་དེ་ལ་ཁྱབ་བྱས་པའོ། །​ སེམས་ཀྱི་བྱུང་བའི་ས་མེད་པས། །​ བར་དོ་འགྲོ་བའི་ལམ་འགགས་པ། །​ ཐམ་རྟོགས་ཚོགས་འབྲས་བུ་སྟོར། །​ སྙེད་པ་མེད་པའི་དབྱིངས་སུ་ཐིམ། །​ དེ་ལ་ཙལ་བས་སྙེད་མི་འགྱུར། །​ མི་སྙེད་ཆོས་ལ་བརྗོད་པ་ཆད། །​ དེ་ནི་བྱང་ཆུབ་མྱུར་གུ་གཅད། །​ བཅད་བྱའི་ཡུལ་དང་ཤེས་བྱེད་མཐའ། །​ གཉིས་འཛིན་བྲལ་བའི་ངང་དུ་ཆོད། །​ དེ་ཡི་འབྲེལ་པ་ཟད་པས་སྟོངས། སྟོང་པའི་ཆོས་ལ་བཅད་གཞི་ཡས། །​ ཡས་པའི་དི་ལ་སུས་དྲིས་པས། །​ སངས་རྒྱས་འབུམ་ཕྲག་གྲངས་མེད་པའི། །​ ཞལ་མཐོང་འདི་དང་གཉིས་སུ་མེད། །​ བྱས་མེད་སྣང་བའི་རྩལ་ཆོད་པས། །​ ང་ནི་སྤྱད་བྱའི་རྟོགས་པ་འགགས། །​ དེ་ནི་མི་འགྱུར་ལམ་ཟིན་པས། །​ སངས་རྒྱས་རྣམས་ཀྱི་མ་བྱས་པའོ། །​ ཡང་ཉོན་སེམས་དཔའ་ཆེན་པོ་ཁྱོད། །​ དེ་ལ་ངེས་པ་མ་ཐོབ་ན། །​ ཕྱི་ནང་འདུས་པའི་སྣང་བ་ལ། །​ རང་དབང་བྱས་ལ་ཕྱོག་གནད་བསྒྱུར། །​ འདི་ལྟར་བྱས་པས་སྤྱོད་པ་ནི། །​ སེང་གེ་བུ་བཞག་ལྟ་བུ་ལ། །​ ཕྱོགས་དང་འཛིན་པ་མེད་པར་རྒྱུ། །​ འདི་ལ་ལས་དང་བག་ཆགས་ཀྱིས། །​ ཉོན་མོངས་འགྱུར་བའི་རྟ་ཆད་ནས། །​ འགྲོ་དང་འོང་པའི་ལམ་དང་བྲལ། །​ བེམ་པོ་འདུབ་པའི་གཟུགས་རྣམས་ལ། །​ བླང་བའི་བྱ་སྟེ་བཞག་པ་སྨིན། །​ དེ་ལ་སེམས་ཀྱི་མ་ཡངས་ན། །​ སེམས་འདུས་རྟོག་ཁྲེག་ཀྱིས་ཆོད། །​ དྲན་པ་འདུས་བའི་ཕུང་པོ་གང་། །​ དེ་ཉིད་གང་བྱུང་སར་གཞག་ན། །​ སྡུག་བསྔལ་འདུས་པའི་ཕུང་པོ་ཞིག། སླར་ལ་བླང་པའི་ས་མེད་པ། །​ ཡང་ནི་མིག་གིས་གཟུགས་ཤེས་ན། །​ སྟོང་པའི་ཆོས་ལ་འདིས་ལོངས་སྤྱོད། །​ དེ་ལྟར་འདོད་ཡོན་བརླབས་བཅད་ནས། །​ རྫོགས་པས་འདི་ལས་བཤད་དུ་མེད། །​ སྨྲ་མེད་བསམ་པའི་གཞི་ཟད་པས། །​         དུས་གསུམ་འབྲལ་བ་མེད་པ་ཡི། །​ གྲོག་འདི་གཞན་དང་མཉམ་པ་མིན། །​ གོང་ན་མེད་པའི་སྒྲ་དང་དྲི། །​ རོ་དང་རེག་པའི་དངོས་པོ་སྟོར། །​ རྙེད་པའི་མཆོག་སྟེ་དགོང་འདོད་ཕྲད། །​ འདི་ནི་འཆོས་མེད་སེམས་ཟད་པས། །​ སེམས་འདུས་ཡིད་ཀྱི་སྤྱོད་པ་ལས། །​ འདི་ནི་ཅུང་ཟད་ཙམ་ཐལོ། །​ རྒྱལ་བས་འཕམ་པས་གསལ་ནས། །​ བྱས་ནས་བྱུང་བ་དུམ་བུར་མཆོད། །​ འདིས་ནི་འཁོར་བའི་མཐའ་མ་ཆོད། །​ དེ་ཆོད་ཡུལ་ལ་ཕར་སྤྱོད་པས། །​ སྦྱང་བྱ་སྤྱོང་བའི་གཞི་མེད་པས། ཐེག་པ་རྣམས་ལས་དྲིས་པ་མེད། །​ འདི་ནི་རིང་ཐུང་ག་འདུས་ཏེ། །​ སེམས་ཀྱི་ཡུལ་དེ་བརྩལ་བས་ནི། །​ སྡུག་བསྔལ་རྣམས་ཀྱི་རྩ་བ་ཆོད། །​ བྱ་བ་མ་ཡིན་དུ་མཚུངས་པས། །​ འཁྲུལ་གཞི་དག་པའི་མེ་ལོང་མཐོང་། །​ དེ་མཐོང་སྡེ་གནོད་ཆོས་རྣམས་ཤེས། །​ ལོ་རྒྱུས་ཆེན་པོ་རྨད་བྱུང་བས། །​ མ་བྱས་ཐོས་བསམ་གཞི་ལ་རྫོགས། །​ སྒྲིབ་བྲལ་བདེ་བའི་ལུས་ཐོབ་ནས། །​ འཇིག་པ་བརྒྱད་ལས་འཇིག་མེད་པར། །​ སླར་ན་གཞན་གྱིས་ལས་རྣམས་འཇོམས། །​ འདི་ལྟར་སྤྱད་པར་འབྲལ་མེད་པའི། །​ མཐའ་ལ་ཆུད་ན་ལྡོག་པ་མེད། །​ སྤྱོད་པས་ཕ་རོལ་བདེ་ཐོབ་ནས། །​ སྡུག་བསྔལ་ཕྱག་མས་འདེབས་པའོ། །​ བསྐལ་པར་བསགས་པའི་ལས་ཀྱི་མཐའ། །​འདིར་སྤྱད་གཞན་ལ་མ་ལྟོས་པས། །​རང་གི་ཤུག་ལ་རང་ས་ཟིན། །​འདི་ལ་གང་གིས་རླགས་མ་རྙེད། །​སྲ་བའི་གོས་གོན་མགོན་ནུས་ཟད། །​འཁོར་དང་འདས་བའི་འབྲེལ་གཅད་པས། །​སྔོན་གྱིས་ཡུལ་ཡང་ད་ལྟ་མཐོང་། །​རྡོ་རྗེ་འཛིན་པ་ཞེས་བྱ་བ། །​སྔགས་ཀྱི་ཐེག་པའི་མཐར་ཐུག་པའོ། །​ལམ་འབྲེལ་དུས་ཀྱི་གནས་ཤེས་པས། །​འདི་ལ་སྤྱད་དང་སྤྱོད་པས་སྟོངས། །​རང་བཞིན་མི་གནས་སོས་དལ་བས། །​ང་བདག་རྣམས་ཀྱི་ཁོག་པ་སྟོངས། །​འཁོར་རྐྱེན་རྒྱུ་འབྲས་རྩ་བ་རུལ། །​འདི་ལ་ལོ་འདབ་སྐྱེ་མི་སྲིད། །​དེ་ན་ངས་ནི་ང་རང་མཐོང་། །​སངས་རྒྱས་རྣམས་ཀྱི་སྤྱོད་པ་ནི། །​བྱ་བའི་གཞི་མེད་བྱེད་མེད་པས། །​འདི་ལ་འདས་པའི་ལམ་བྲལ་ཏེ། །​འབྲས་བུ་རྣམས་ཀྱི་རེ་གནས་པ། །​སྟོང་མིན་སེམས་ཀྱི་གོམས་སྤྱད་ནས། །​ངོ་ཤེས་འདི་ལ་འདོད་བྲལ་བ། །​ནོར་མེད་ལོངས་སྤྱོད་ཆེན་པོ་ལ། །​བྱ་བྲལ་གནས་པའི་སྙིང་པའོ། །​བལྟས་པའི་དུས་གསུམ་དུས་གཅིག་གསང་། །​ལྟ་བའི་ཡུལ་མཐོང་ལྟ་མཁན་མཐོང་། །​སྔོས་བཅས་རང་དག་ཟག་པ་མེད། །​འབྱུང་བའི་ལུས་ཟད་ཉོན་མོངས་སྟོངས། །​དེ་ཡི་ཡུལ་ལ་བསམ་མི་ཁྱབ། །​བསམ་མི་ཁྱབ་པས་བསམ་མེད་ལ། །​བྱས་མེད་བྱུང་མེད་འགྲོ་ས་མེད། །​ང་མེད་གཞན་མེད་འཁྲུལ་བ་མེད། །​མཐོང་དུ་མེད་པའི་གཞི་ལ་མཐོང་། །​རྒྱུན་ཆད་སྐྱེན་འགག་འབྲས་བུ་ཀླག།འགྲོ་འོང་ལས་ཀྱི་འཁྲུལ་ཕྱུག་མེད། །​དེ་དོན་སེམས་ལ་རང་གཞག་པའོ། །​དེ་ལྟར་གསུམ་ཤེས་གཉིས་མེད་ལ། །​གཞི་སོགས་གྲངས་ཀྱི་མཐའ་མེད་པས། །​གསུམ་ལ་འཁྲུལ་ས་རང་གཞག་པས། །​འགྱུར་ཞིང་འཕྱུག་པ་མེད་པར་བསྡམས། །​འཁྲུལ་དུ་མེད་པས་ཅིག་ཤེས་ལ། །​གཉིས་སུ་འཛིན་པ་མི་སྲིད་པས། །​འཁྲུལ་ཞིང་འཁྲུལ་བའི་སྡེང་གཞག་པས། །​བྱ་བྱེད་ཐམས་ཅད་རྒྱུན་ཆད་དོ། །​སེམས་པའི་ས་ནི་ཡང་གསོལ་བ། །​འཁོར་བའི་སྙིང་པོ་དེ་ལྟར་ན། །​རང་བཞིན་ཇི་ལྟར་རྣམ་པར་དག།འདི་ཉིད་སྟོན་པས་ཡང་གསུངས་གཅིག།་དེ་ནས་རྡོ་རྗེ་འཆང་བ་ཡིས། །​རང་བཞིན་གནས་པའི་ཚིག་འདི་གསུངས། །​གསང་བདག་སེམས་དཔའ་ཡར་ཉོན་ཅིག།འདི་ལྟར་སྣང་དང་སྟོང་པ་ཡིས། །​བསྡུས་པའི་ཅི་ཙམ་པ། །​རང་བཞིན་གྱིས་ནི་རྣམ་པར་དག།འབད་རྩོལ་དྲི་མ་ལས་འདས་པས། །​སྤྱོད་བྲལ་སྒོམ་དག་ལྷ་ཡུལ་སྟོངས། །​མ་བྱས་རང་གི་སར་བཞག་པས། །​དེ་དག་རང་དག་བཞག་ཡས་དག།རྒྱུ་དག་རྐྱེན་དག་འབྲས་བུ་དག།ཆོས་དག་སེམས་དག་འཁྲུལ་བ་དག།དངོས་དག་ཡུལ་དག་འགྱུར་འབྲེལ་དག།དག་ནས་སེམས་ཀྱི་སྣང་དག་པས། །​དེ་རང་བཅད་པར་བྱ་བ་མ་ཡིན་ནོ། །​དེ་ལྟར་དག་པའི་སྣང་བ་ལ། །​ སེམས་སྤྱད་ལམ་དུ་བླངས་པ་ཡིས། །​ས་ཆུ་མི་རླུང་དག་པར་སྣང་། །​དག་པའི་ལུགས་དང་དག་པ་ཡིས། །​དག་པའི་ཡུལ་དང་དག་པའི་དུས། །​དག་པའི་ཚད་དང་དག་ལ་གནས། །​དེ་ལྟར་བེམས་པོའི་ལམ་དུ་ཡང་། །​སུ་ཡིས་སློང་བར་ཤེས་པ་ནི། །​བསླངས་པའི་ལོངས་དང་ལོངས་ལུགས་དང་། །​ལེན་པའི་དུས་དང་ལོངས་པའི་ཚད། །​བསླངས་པའི་འབྲས་བུ་སྐྱ་དང་སྐྱེས། །​ངེས་པར་སྐྱེས་པ་རྣམས་ནས་ནི། །​ཕྱི་ཡི་སྣང་བ་ཆོས་ཀྱི་སྐུར། །​གནས་པ་མཐོང་བར་འགྱུར་བས་ན། །​མཐོང་བ་དེ་ལ་གང་གོམས་པ། །​སྐུ་གསུམ་རྒྱལ་བའི་ཞིང་ཁམས་མཐོང་། །​འདི་ནི་མཐོང་བའི་ལམ་ལ་གནས། །​འབྲེལ་པ་དག་པ་ངེས་ཐོབ་པས། །​དངོས་སུ་སྤྱོད་པ་ཉིད་ལས་སོ། །​ནང་གི་སེམས་དང་སེམས་ལས་བྱུང་། །​སེམས་ཀྱི་སྣང་བ་གཉིས་དག་ལ། །​རང་སར་བཞག་དང་རང་གྲོལ་དང་། །​རང་བྱུང་རང་ཤར་རང་གིས་ཡུལ། །​རང་གི་རྩེ་ལ་རྩ་བ་བཅད། །​རང་གནས་རང་གི་ཐོབ་པ་ལ། །​སེམས་རྒྱུན་སྟོངས་དང་མེད་པ་དང་། །​འགྲོ་འོང་འབྱུང་འཇུག་ཤེས་པ་ཡིས། །​འགྱུ་བ་ཡལ་བའི་བར་ཕྱེ་དང་། །​གཟུང་དང་བཏང་དང་གློད་པས་ཤེས། །​སྦྱོར་བ་དང་ཡང་འབྲལ་བ་དང་། །​སྟོར་དང་བསྡུ་བའི་གནད་བཏུན་ནས། །​སེམས་ཀྱི་མེ་ལོང་གཟུགས་བརྙན་འདྲ། །​སྣང་ཡུལ་བཟུང་བར་མེད་པ་ནི། །​ཡེ་ནས་དེ་ལྟར་གནས་བ་ལ། །​ཤར་བ་རྣམས་ཀྱི་ཐོག་མ་ལ། །​བྱུང་བ་རྣམས་ཀྱིས་བར་གསལ་བས། །​ཤེས་པ་རྣམས་ཀྱི་ཐ་མར་སྣང་། །​དེ་ཡང་འབྲས་བུའི་དུས་ལས་ནི། །​གནད་ལ་ཕབ་ངེས་བཟུང་བྱའོ། །​ཕྱི་ནང་བཅུད་དང་བར་བཀྲུགས་དང་། །​ནང་དུ་གནད་ལ་བོར་བ་དང་། །​གང་ཡུལ་གང་མཐོང་གནས་བ། །​གང་གི་མཐོང་དང་གང་གི་ཚོར། །​གང་ལ་བཤད་དང་གང་ཐོབ་དང་། །​འཁྲུལ་པ་རྣམས་ཀྱི་ཐ་མ་ལ། །​འཁོར་བའི་ཐོག་མའི་ཤེས་པས་གསལ། །​སྣང་བྱེད་བསྡུ་དང་རྨུགས་པ་དང་། །​ཡུལ་ལས་དཀར་དང་གནས་སུ་བཞག། ལམ་དུ་བཞག་དང་གཞི་ལ་བསྐྱལ། །​བསྟས་མཐོང་མེད་པས་སྣང་བ་ལ། །​བསྒྱུར་དང་བཅད་བའི་ལས་རྣམས་འགྲུབ། །​ཟས་དང་གོས་དང་འགྲོ་བའི་རྟ། །​བཅུད་དུ་གྲིལ་བས་ཤེས་བས་ན། །​འགྱུར་གྱུར་ཐོབ་རྣམས་ལས་ཉིད་དང་། །​ཡང་དག་པའི་ལས་ཀྱི་མཐའ། །​ཁྲེགས་ཀྱི་བཅད་པས་བྱ་བ་རྫོགས། །​རང་གི་དཔའ་དང་གཞན་གྱིས་བགེགས། །​ཚད་ལྡན་བླ་མས་མཁྱེན་པ་ལ། །​ལེགས་པར་བྲིས་གཅིག་དད་སེམས་ཅན། །​འདིས་ནི་འཁོར་བའི་ཟད་དུ་ཤེས། །​བྱ་བ་རྫོགས་པའི་རྣལ་འབྱོར་པ། །​དེ་ནས་གསང་བདག་སེམས་དཔའ་ཆེ། །​སྟོན་པ་ཉིད་ལ་ཡང་བསྟན་པ། །​ཀྱཱཻ་ཀྱེ་སྟོན་པ་རྡོ་རྗེ་འཆང་། །​རང་གི་གཞི་དག་པ་དེ་ལྟར་ན། །​སྙིང་པོ་ལམ་དུ་ཇི་ལྟར་བསླང་། །​འདི་ལྟ་བདག་ནི་ཞུ་བ་བགྱིད། །​སྟོན་པས་གསུང་པ་སེམས་ཅན་ཆེ། །​ཡང་ནི་འདི་ལྟར་ཡ་མཚན་ཅན། །​སྙིང་པོ་ཐམས་ཅད་འདུས་པའི་གནད། །​སངས་རྒྱས་གཞན་གྱིས་མ་གསུངས་པའི། །​དོན་འདི་ངས་བསྟན་ཁྱོད་ཀྱི་ཉོན། །​སྤྱི་དང་རང་གི་ཁྱད་པར་ལས། །​ཐོག་པ་རྣམས་ཀྱི་ཐེ་ཚོམ་ས། །​འདི་ལས་བྱས་བྱུང་མེད་པ་རྟོགས་པའི། །​སེམས་ལས་གྱུ་ཡང་ཟད་ན། །​སེམས་ཀྱི་འགྱུ་ས་གཞིག་པ་དང་། །​དེ་ཡིས་རྗེས་ལ་གར་སོང་གནས། །​སྙིང་པོ་འདུས་པས་སྤྲོས་ཟད་པ། །​ཚིག་ལ་གནས་བ་མེད་པ་ནི། །​གནས་པའི་དུས་ན་གློད་པ་ཡིས། །​རླུང་ལ་མཁས་བ་ཁོ་ནའོ། །​དྲན་པ་འགྱུས་པའི་སྙིང་ནི། །​རང་སོར་གཞག་པ་ཁོ་ནའོ། །​ཡུལ་རྣམས་བསྡུས་པའི་སྙིང་པོ་ནི། །​ཐད་ཁར་བཅད་པ་ཁོ་ནའོ། །​གནས་ལུགས་བསྡུས་པའི་སྙིང་པོ་ནི། །​སེམས་བྱུང་བཙལ་བ་ཁོ་ནའོ། །​གྲོལ་ལུགས་བསྡུས་པའི་སྙིང་པོ་ནི། །​།བྱས་པ་མེད་པ་ཁོ་ནའོ། །​ཤར་ལུགས་རྣམས་ཀྱི་སྙིང་པོ་ནི། །​བལྟས་པས་མཐོང་བ་ཁོ་ནའོ། །​དེ་ལྟར་དེ་ནི་འབྲས་བུའོ། །​ལམ་ནི་ཆེད་དུ་བྱེད་པ་དང་། །​ཤུགས་ལ་བྱུང་བ་ཉིད་དང་ནི། །​རང་བཞིན་གནས་པ་ཉིད་ན་དགོ།ཆེད་ལ་ཡང་ན་བྱི་བས་བྱ། །​ལུས་དང་བག་དང་ཡིད་ཀྱི་ཡང་། །​གནད་དང་དེའི་བྱ་བ་བླ་མའི་ལུང་གི་ཤེས། །​ལས་ནི་འདུག་དང་ལངས་པའི་ཁྱད་བར་ལས། །​བྱ་བ་དང་ནི་བྱེད་པའི་ལས་ལ་བརྩོན། །​ངག་ནི་ཨ་ལི་ཀ་ལི་མཉམ་སྦྱོར་བས། །​སྒྲ་དབྱངས་ཚིག་གི་འབྲེལ་གྱི་སྦྱར། །​ཡིད་ནི་འཕང་དང་དགུག་དང་བསྒྱོག་པ་ནི། །​སོ་སོའི་དུས་དང་སྦྱར་བས་རྒྱལ་བའི་དགོངས་པ་འགྲུབ། །​ཤུགས་འབྱུང་ལམ་ནི་བཞག་པ་ཙམ་གྱིས་འགྲུབ། །​དེ་ལ་འབད་ཅིང་རྩོལ་བ་རླུང་གི་སྒོ་དགག།རང་བཞིན་གནས་པའི་ལམ་ནི་འབྱུང་འགྱུར་དེ། །​གཡོ་དང་འགྱུར་དང་བསྒྱུར་བའི་ལས་མཐའ་ཅན། །​དེ་ལ་རང་གི་བྱ་བ་མེད་པས་ལེགས་བཞག་པའོ། །​སློང་བའི་དུས་དང་བསླང་དང་བྱ་བའི་གནད། །​འགྱུར་བའི་སེམས་ལ་མཚན་མེད་ཆོས་གསལ་བ། །​གཡོ་དང་འབྱལ་དང་འཕྲིག་པའི་དུས་ཀྱི་ཤེས། །​འདི་ལ་རང་བཞིན་བཞག་དང་བྱས་པའི་གནད་གཉིས། །​ལམ་ལ་བྱུང་དང་བདེང་དུ་ཚུད་པ་ལ། །​རྟོག་པའི་ལུས་དང་དེ་ཡིས་ཐ་ཡི་ཕྱེ། །​གཉིས་སུ་ཡོད་དང་དེ་ཡི་ཐ་གསལ་བ། བར་དང་དང་པོ་ཉིད་ལས་གསལ་ཕྱེ་བ། །​དེའི་ལུས་དང་ཡན་ལགས་སོ་སོའི་དུས། །​འདི་ལྟར་སྦྱར་རོ་རྣལ་འབྱོར་སེམས་དཔའ་ཆེ། །​རིག་པ་གནས་པས་དེ་ལ་ཉེས་སྤྱོད་པ། །​ལུས་ལ་གནས་དང་ཡུལ་དང་འགྱུ་བ་ལས། །​རབ་དང་འབྲིང་དང་ཐ་མ་ཡི། །​འཇུག་པའི་རྣལ་འབྱོར་པས་མཐོང་ན། །​གང་ལ་མཐོང་བ་ཙམ་གྱིས་འཁྲུལ་པའི་འཁོར་རྒྱུན་གཅོད། །​རང་གི་གནས་པ་མིག་གི་བྱེད་པ་ཅན། །​མངོན་གསུམ་པ་དང་དེ་ཡིས་འགྱུར་ཤེས་ནི། །​གོམས་པའི་ཤུགས་ནི་རང་གིས་ལས་སོ། །​ཡང་ནི་རྫོགས་སངས་རྒྱས་པ་ལ། །​གནས་པ་ཙམ་གྱིས་ཐིག་ལེའི་དྲིལ་ལུགས་ཞུས། །​སྟོན་པས་གསུངས་པ་འདི་ལ་ལེགས་ཉོན་ཅིག།ཆོས་ཉིད་གནས་པ་ཙམ་ལས་རང་བྱིན་བརླབ་པ་ནི། །​འཁྲུལ་པའི་འདུ་བྱེད་ཁྱད་པར་གནས་པ་ལ། །​རྒྱ་ཡི་གདབ་དང་ཁུང་སུ་བཙུད་པ་ཡིས། །​དེ་མ་ཡིན་ལས་དེའི་སྣང་ཤར་བ། །​དེ་ལ་ལུག་གནད་སེམས་གནད་མཐའ་ལ་གནས་པས་བདག།འཁོར་དུས་འབྲེལ་དུས་རང་ལ་གནས་བས་སྤྱད། །​གོང་མོའི་ཟླ་མཐོང་གསལ་བས་ཆེ་བ་དང་། །​དང་པོ་ལས་རྩིས་གྲངས་ཀྱི་ཆེ་བ་དང་། །​ཡེ་ནས་གནས་བས་རང་བཞིན་ཆེ་བ་དང་། །​སྔ་མ་ལས་འགྱུར་དོན་གྱི་ཆེ་བ་ཉིད། །​རང་རིག་མཐོང་དང་གཞན་ལ་བསླབས་པ་ཡིས། །​གོང་མའི་བདེ་ནས་འོག་མའི་ཟླ་དང་བྲལ། །​ཐད་ཀ་དུས་ཀྱི་འགྱུར་བ་ངེས་ཕྱེ་བས། །​རྫོགས་ནས་དེ་ཡིས་དོན་ལས་ཆེན་པོར་གསལ། །​འགྱུ་དང་ཡུལ་ལ་མི་གནས་ཆེན་པོ་ཉིད། །​བདེ་བ་འདོད་བྲེལ་ནམ་ཁའི་ཡུལ་གནས་པ། །​འདི་ནི་རྫོགས་པ་ཆེན་པོའི་རྣལ་འབྱོར་དུ། །​སྤྲོས་དང་ཟུག་རྔུ་བྲལ་བའི་ཐིག་ལེ་ལ། །​འདི་ནི་གཅིག་དུང་གནས་པའི་ཟླ་དང་ཆར་ཤེས་དང་། །​གཅིག་ཏུ་སྡོམས་པས་དུ་མ་ཆོད། །​འདི་ལ་མིང་གི་ཐ་རྣམས་ཟད། །​འདི་ནི་ཉག་ལས་གཅིག་ཏུ་དྲིལ། །​སྤྲོས་པ་མེད་བའི་བདེ་ཆེན་གནས། །​རྩ་ཡི་གནས་ནས་འཕྲ་འགྱུར་འཁྲིག།སྒྲོན་མ་ལས་ནི་དངོས་སུའོ། །​ཡུལ་ལ་ཟླ་དང་བཅས་པ་ཉིད།སྔོན་དུ་གྱུར་པས་འཕེལ་བའི་རྫོགས། །​འདི་ལ་ཟད་པས་ལས་ཟད་པས་འགྱུར་བའི་རླུང་དེ་ཁྲིག་བ་ཐིག།ཁྱབ་པ་དངོས་པོ་ཐིག་ལེའི་ཡུལ། །​འདི་ལས་མངས་དང་ཉུངས་བ་དང་། །​འཕེལ་འགྲིབ་རྣམས་ཀྱི་དང་པོ་ལ། །​བྱ་བ་བྲལ་བའི་སྔོན་དུ་སྟོས། །​འདི་ནས་ཆེད་དང་ཡས་པ་དང་། །​འབྱམས་དང་རླུང་དུ་གྱུར་བ་ལས། །​སངས་རྒྱས་ཀུན་གྱིས་ཡུལ་མཐོང་བའོ། །​སྒྲོན་མ་སྟོན་པས་རྣམ་པར་འགྱུར། །​སེམས་ཀྱི་བྱ་བ་ལས་འདས་ནས། །​བདེ་དང་སྡུག་བསྔལ་གཅིག་བྲལ་ནས། །​ང་དང་རང་བཞིན་འོད་དང་ནི། །​ཁ་དོག་སོ་སོར་གསལ་བ་ལས། །​རིག་པའི་དབྱིངས་དང་འབྲེལ་བས་ན། །​འདི་ལ་རྣལ་པར་ཕྱེ་བ་ཡིས། །​གཞི་ཡི་བྱེ་འགྱུར་མིག་དང་འབྲེལ། །​ཁུང་ནི་གཞི་ལ་བྱུང་བ་ལ། །​ད་ལྟ་སེམས་ཅན་ལུས་ལ་གནས། །​ལམ་ནི་འཁྲིག་པ་རྩ་ལས་ཡང་། །​དུ་མ་རྩེག་བའི་རང་བཞིན་ཅན། །​འདིས་ནི་འཁོར་བའི་མུན་པ་སེལ། །​དང་པོ་མེད་པས་ཡོད་པར་སྣང་། །​བར་དུ་འོད་དང་ཐིག་ལེ་ཉིད། །​ཐ་མ་སྐུ་དང་ཡེ་ཤེས་སོ། །​དེ་ལས་སྟོན་བར་གྱུར་པ་ལས། །​ཟད་པ་རྣམས་ཀྱི་འབྲས་བུ་སྟེ། །​འཁོར་བའི་ཐོག་མཐའ་དེས་ཆོད་ནས། །​རྫོགས་པའི་རྩེ་མོ་མཐར་ཕྱིན་པས། །​དེ་ལས་ཕྱོགས་མེད་ཐུགས་རྗེས་འཕྲོས། །​ཀུན་དོན་མ་ལུས་རྫོགས་བྱས་ནས། །​རང་གི་ས་ལ་གནས་པས་སྟོང་། །​ཅིག་ཆར་བ་དང་རིམས་ཀྱིས་སོ། །​རྡོ་རྗེ་འཆང་ལ་ཡང་ཞུས་པ། །​རྩེ་གཅིག་བ་ལས་ཆོས་ཉིད་ལམ། །​ཁྲེགས་ཀྱི་གཅོད་པ་ཇི་ལྟར་ལགས། །​ཐུགས་རྗེ་ཅན་ལ་བདག་ཉིད་ཅན། །​དེ་ནས་སྟོང་པ་རྡོ་རྗེས་འཆང་། །​ངེས་པའི་འབྲས་བུ་ལམ་ཞུགས་ཕྱིར། །​རྩེ་གཅིག་འདུས་པའི་དགོངས་པ་གསུངས། །​ཆོས་ཉིས་རྕེ་མོ་དལ་བ་ཡིས། །​འཁོར་བ་འདུས་པའི་རྟོག་པ་གཅོད། །​རང་སེམས་འདུས་པའི་རྩེ་མོ་ཡིས། །​འདས་པའི་མཉམ་པའི་དོན་རྣམས་གཅོད། །​རང་གནས་འདུས་པའི་རྩེ་མོ་ཡིས། །​ཆོས་རྣམས་ཟད་པར་གཅོད་པར་བྱེད། །​སྐུ་མེད་རང་དག་རྩེ་མོ་ཡིས། །​བྱ་བྱེད་འབད་བརྩོན་ཐམས་ཅད་གཅོད། །​རྩེ་མོའི་དངོས་ལ་འོད་འཕྲོས་པས། །​འཁོར་འདས་གཉིས་ཀ་དུས་གཅིག་ལ། །​སྟོང་པ་རྩེ་མོ་གཅིག་པ་ལ། །​རྟོག་དང་མ་རྟོགས་དུས་གཅིག་མཐོང་། །​གནས་པའི་ལས་མཐའ་རྫོགས་པའོ། །​ལུས་རྫོགས་རྕ་རྣམས་འདུས་པ་ཡིས། །​ལུས་ལ་གནས་དང་འཕེལ་དང་ཉམས། །​འགྱུར་དང་འཇིག་དང་འཕོ་བར་བྱེད། །​ཆོས་ཉིད་ཡི་ཤེས་འཕེན་པ་དང་། །​འཕང་དང་འཙིན་པ་དང་། །​འགྲེལ་དང་གོང་འོག་སྡེབ་རྣམས་བྱེད། །​དག་ལ་སྒྲ་དོན་འདུས་བ་ཡིས། །​ཡིན་དང་མིན་དང་ཡོད་དང་མེད། །​མཐོང་དང་མ་མཐོང་ཤེས་མ་ཤེས། །​གཉིས་ལྟོས་རྣམས་ཀྱི་བྱེད་པ་ཅན། །​སེམས་ལ་འཁོར་འདས་གཉིས་འདུས་པས། །​སངས་རྒྱས་སེམས་ཅན་རྒྱུ་འབྲས་དང་། །​ཆོས་དང་ཆོས་མེད་རིམ་པ་དང་། །​བདེ་དང་སྡུག་བསྔལ་གཞི་མ་བྱེད། །​དེ་ལྟར་འདུས་པས་རྣམ་པར་གྱེས། །​འདི་ལ་བསམ་གྱིས་མི་ཁྱབ་པའོ། །​གང་ལ་རྟོག་པའི་མཐའ་ཤེས་ན། །​དེ་ལས་ཆོས་ཉིད་ཡུལ་གནས་པས། །​འབྲེལ་པ་མེད་པའི་དོན་ཉིད་ལས། །​ཆོས་དང་གང་ཟག་མཐའ་ཕྱེ་བས། །​འཁྲུལ་པའི་རྟོག་པ་དུས་གཅིག་འགག།ལུས་ཀྱི་མི་རྟོགས་པ་དལ་རྟེན་པས། །​རྟོག་པ་རྣམས་ཀྱི་འབྲེལ་པ་ཆོད། །​ངག་གི་མི་རྟོག་སྨྲས་གཅད་པས། །​བརྗོད་མེད་དོན་གྱི་གཞི་མ་རྙེད། །​སེམས་ཀྱི་མི་རྟོག་རང་སར་བཞག།དེས་ནི་རྐྱེན་རྣམས་ཆོས་ཉིད་འགྱུར། །​གསུམ་སྡོམ་སྐུ་གསུང་ཐུགས་ལ་གནས། །​དེའི་འབྲས་བུ་བདེ་སྐངས་པའོ། །​བསམ་པ་འཕྲོ་དད་འདུས་པ་དང་། །​མཆེད་དང་ཡས་དང་འབར་བ་དང་། །​རྟོག་དང་མི་རྟོག་འདུས་མཆེད་ཅན། །​ཡུལ་ལ་འཇུག་དང་གནས་པ་དང་། །​དེ་ལས་འཕྲོས་དང་མཆེད་པ་ཡིས། །​དྲན་བསམ་རྣམས་ཀྱི་གཞི་མ་བྱེད། །​རང་བཞིན་འགྱུར་དང་བློ་བུར་པས། །​རྐྱེན་རྣམས་དུ་མ་སོགས་པར་བྱེད། །​དེ་ལས་བདེ་སྡུག་ཕུང་པོར་འགྱུར། །​འཁོར་བའི་ས་བོན་འབྲས་བུརོ། །​དེ་ཉིད་བསྟན་པར་འདོད་པ་ཡིས། །​ཐ་མ་ལ་སཽའི་ལས་བཟློག་ཏེ། །​ལུས་དང་ངག་དང་ཡིད་ཀྱི་གནད། །​གཞན་ལ་བཟློག་པ་རྣལ་འབྱོར་པས། །​བྱ་དང་བྱེད་པ་མེད་པ་ལས། །​རང་བཞིན་དུས་ན་ཅོག་པུ་ལ། །​མཚན་ཉིད་གཞི་དང་ལྡན་པས་བྱ། །​གཉིད་ལ་འཁོར་ལོ་ལྟ་བུ་ལ། །​འགྲོ་ན་མཁའ་ལྡིང་གི་ལུགས། །​འདུག་ན་གཡས་དང་གཡོན་དུའོ། །​བྱེ་བྲག་གཞན་ལམ་མཁས་པ་ཡིས། །​ཇི་ལྟར་རིག་པ་སྦྱར་བའི་འགྲུབ། །​ངག་ནི་རང་བཞིན་སྨྲ་དང་བྲལ། །​ཉལ་ན་ཡི་གེ་འཕྲེང་བ་བཟླས། །​འགྲོ་ན་རླུང་གི་ཡི་གེ་རྟེན། །​འདུག་ན་ཨ་ཡི་རྣམས་པ་ལ། །​བྱེ་བྲག་གཞན་ལ་མཁས་པ་སྦྱར། །​ཡིད་ནི་རང་བཞིན་ཐུན་མོངས་ལ། །​གར་སོང་ཕྱི་ནང་རྗེས་རྣམས་གཅོད། །​ཉལ་ན་ཙི་ཏར་འཇུག་པར་རྨིས། །​འགྲོ་ན་རྨི་ལམ་འཁའ་ལ་སྤུར། །​འདུག་ན་ཐིག་ལེ་འདྲ་བར་བཞུགས། །​བྱེ་བྲག་གཞན་ལ་མཁས་པ་ཡིས། །​གང་ལ་མོས་པ་མན་ངག་གིས། །​གནས་རྣམ་སྒྱུར་བ་རྣལ་འབྱོར་པས། །​དེ་ཡིས་དངོས་གཞི་འབྲེལ་སྦྱར་བ། །​འགྱུ་བའི་ཡུལ་དབྱུང་དྲན་པ་གསད། །​བསམ་པའི་ཡུལ་ལྷ་སེམས་རྒྱུན་གཏོར། །​ཡུལ་དང་ཤེས་པ་སྦྱར་བས་ཤེས། །​འཕྲོ་ལས་མཆེད་པའི་དུས་མཐའ་གཅད། །​གང་སྣང་རང་གི་ཐོག་ཏུ་དབབ། །​གཅིག་གི་འབྲེལ་པ་ཅིག་གཅད་ན། །​དེ་ལ་ཉམས་ཀྱི་ཤེས་པ་སྐྱེ། །​དག་པ་ཆེན་པོ་གཞི་ཉིད་ན། །​བདེ་གསུམ་བཅུཡྕ་གཉིས་འཆར། །​དེ་ལས་འབར་དང་བདེ་ལ་སྐྱེ། །​ཐོབ་པས་འཁོར་ལོ་ཅན་ལ་འཇུག།འགྲོ་དང་འོང་པའི་ཤུགས་རྒྱུན་འགག།དེ་ལ་སེགས་ཀྱི་རྣམས་འཕེལ་བས། །​ཕར་དང་གཞན་གྱི་བྱ་བ་བྱེད། །​ལོག་དང་མ་ལོག་སྤོར་གནས་དང་། །​ཇི་ལྟར་འགྱུར་བ་གྲངས་ལས་ཤེས། །​བསོད་ནམས་ལས་ཀྱི་བྱེ་བྲག་རྣམས། །​ཇི་ལྟར་འདོད་པ་ཤེས་པ་དང་། །​འཆི་འཕོ་སྐྱེ་བའི་གྲངས་ཉིད་ཀྱང་། །​འཕེན་དང་སྡུག་པ་ལས་ཤེས་པ། །​དེ་ཡི་རང་བཞིན་རྣལ་འབྱོར་པ་ཡིས་ཡང་། །​ཡིད་གཞུང་དྲན་པ་གསལ་པོ་ཡིས། །​རྒྱུན་དུ་རྩིས་པས་མངོན་པར་གསལ། །​བསམ་གཏན་ཉིད་ཀྱི་ཡོན་ཏན་ནོ། །​ཞེས་བཀོད་པ་ཆེན་པོ་ཨ་ཏི་རྫོགས་པའི་རྒྱུད་ལས། །​ ཁྲེགས་གཅོད་ལས་ཀྱི་མཐའ་གཅོད་པའི་སྐབས། །​ སྐལ་པ་དུམ་བུའི་རྒྱུད་ལས། །​ མན་ངག་ཏུ་ཕྱུང་བ་ཁྲེགས་བདུན་པ་ཞེས་བྱ་བ་རྫོགས་སྷོ། །​ དགེའོ། །​ བཀྲ་ཤིས། །​
